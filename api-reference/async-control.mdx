---
title: "Async control"
description: "Retry, debounce, and throttle helpers from bytekit/async."
---

## Exports

- `retry`
- `debounceAsync`
- `throttleAsync`

## Retry example

```ts
import { retry } from "bytekit/async";

const result = await retry(fetchData, {
  maxAttempts: 5,
  baseDelay: 1000,
  backoff: "exponential",
});
```

## Debounce example

```ts
import { debounceAsync } from "bytekit/async";

const search = debounceAsync(async (query: string) => {
  const res = await fetch(`/api/search?q=${query}`);
  return res.json();
}, 300);
```
